#include <stdio.h>

int changePointerValue(int* test){
    *test = *test + 10;
}

int add30(int test){
    return test + 30;
}

int main(){
    int a = 5;
    int b = 5;
    int k = 20;
    int* g = &k;
    int* c = &a;
    int* d = &a;
    int* e = &a;
    int* l;
    l = e;
    *c = 10;
    *d = 20;
    e = &b;
    *e = 8;
    a = 7;
    e = d;
    *e = 12;
    int f = *c;
    *d = *g;
    changePointerValue(d);
    printf("%i, %i, %i, %i, %i, %i, %i", a, b, *c, f, *d, *l, add30(a));
}
